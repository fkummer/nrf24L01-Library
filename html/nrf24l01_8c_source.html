190  char setup;

224 void __ISR(_EXTERNAL_1_VECTOR, ipl2) INT1Handler(void){
225  //_LEDRED = 1;
226  nrf_read_reg(nrf24l01_STATUS, &status, 1); // read the status register
227  // check which type of interrupt occurred
228 
229  if (status & nrf24l01_STATUS_RX_DR) { // if data received
230  nrf_read_payload(&RX_payload);
231  received = 1; // signal main code that payload was received
232  status |= nrf24l01_STATUS_RX_DR; // clear interrupt on radio
233  nrf_flush_rx();
234  }
235  // if data sent or if acknowledge received when auto ack enabled
236  else if (status & nrf24l01_STATUS_TX_DS) {
237  //_LEDYELLOW = 1;
238  sent = 1; // signal main code that payload was sent
239  status |= nrf24l01_STATUS_TX_DS; // clear interrupt on radio
240  } else { // maximum number of retransmit attempts occurred
241  //_LEDRED = 1;
242  error = 1; // signal main code that the payload was not received
243  status |= nrf24l01_STATUS_MAX_RT; // clear interrupt on radio
244  }
245 
246  nrf_write_reg(nrf24l01_STATUS, &status, 1);
247  mINT1ClearIntFlag();
